使用递归的方式实现

```java
class Solution {
    public List<Integer> spiralOrder(int[][] matrix) {   
        if (matrix.length == 0) {
            return new ArrayList();
        }
        int startRow = 0;
        int startColumn = 0;
        int endRow = matrix.length - 1;
        int endColumn =  matrix[0].length - 1;

        List<Integer> result = new ArrayList<>();
        move(matrix, result, startRow, endRow, startColumn, endColumn);
        return result;
    }
    
        private void move(int[][] matrix, List<Integer> result, int startRow, int endRow, int startColumn, int endColumn) {
        if (startRow > endRow || startColumn > endColumn) {
            return;
        }
        // 向右
        int i = startColumn;
        while (i <= endColumn) {
            result.add(matrix[startRow][i++]);
        }
        // 向下
        i = startRow + 1;
        while (i <= endRow) {
            result.add(matrix[i++][endColumn]);
        }
        // 向左
        i = endColumn - 1;
        while (endRow > startRow && i >= startColumn && i >= 0) {
            result.add(matrix[endRow][i--]);
        }
        // 向上
        i = endRow - 1;
        while (endColumn > startColumn  && i > startRow && i >= 0 ) {
            result.add(matrix[i--][startColumn]);
        }
        // 递归处理内层
        move(matrix, result, startRow + 1, endRow - 1, startColumn + 1, endColumn - 1);
    }
}
```
